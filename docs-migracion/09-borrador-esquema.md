# 09 — Borrador del esquema PostgreSQL — **v2** (para aprobación final)

**Estado: BORRADOR v2 — incorpora las 9 decisiones del usuario (2026-07-04). Nada se aplica a la BD hasta la aprobación explícita del script final.**
La sección de `documento`/`documento_detalle` queda **A VALIDAR contra el sistema de Gestión (Oracle 11g XE)** en cuanto haya credenciales (`GESTION_DB_*` en `.env`): se alineará con esa estructura probada y con sus triggers/procedimientos.

## A. Decisiones aplicadas (todas confirmadas por el usuario)

| # | Decisión |
|---|---|
| 1 | `date` puro cuando el concepto es solo fecha; `timestamptz` para registro/alta/logs/auditoría |
| 2 | **`ubicaciones_geograficas` recursiva** (padre_id, niveles ilimitados) con seed de Paraguay; las tablas referencian una sola `ubicacion_id` |
| 3 | `parametro_sistema` (config) · enums cerrados en código · tabla genérica **`entidad`** para listas abiertas con **FK compuesta (lista, codigo)** |
| 4 | Contexto empresa/sucursal en sesión (no se migra VARIABLES_ENTORNOS) |
| 5 | `usuarios` con id numérico + `codigo_usuario` único + bcrypt + bloqueo por intentos |
| 6 | Propiedades con 2 pares precio/comisión (venta, alquiler) |
| 7 | Operación sobre propiedad **o** entidad inmobiliaria completa (ambas) |
| 8 | **`documento` + `documento_detalle`** (clave empresa+tipo+serie+numero); todo movimiento de dinero tiene documento; cobros lo referencian |
| 9 | **Atributos dinámicos** de propiedades: definiciones agrupadas, obligatoriedad por tipo, precargado, administrable |

### Clasificación enum vs tabla `entidad`

- **Enums (cerrados, la lógica depende de ellos):** EstadoOperacion, EstadoCuota, EstadoCobro, EstadoPropiedad, EstadoIngresoEgreso, TipoOperacion, CondicionOperacion, TipoMovimiento, TipoDocumentoCobro, TipoPersoneria, Sexo, Perfil, TipoDato (de atributos).
- **Listas en `entidad` (abiertas, administrables):** TIPOS_PROPIEDADES, TIPOS_ENTIDADES_INMOBILIARIAS, TIPOS_CONTRATOS, TIPOS_FINANCIACIONES, TIPOS_DOCUMENTOS_IDENTIDAD, MOTIVOS_LIQUIDACION, TIPOS_GASTOS, TIPOS_IMPUTACION, TIPOS_DOCUMENTO (comprobantes), GRUPOS_ATRIBUTOS, NIVELES_UBICACION.
- Seed inicial de `entidad`: los valores reales del doc 07 §3.

## B. DDL propuesto v2

Convenciones: PK `bigint GENERATED BY DEFAULT AS IDENTITY`; auditoría en todas las tablas (`usuario_creacion varchar(20) NOT NULL, fecha_creacion timestamptz NOT NULL, usuario_modificacion varchar(20), fecha_modificacion timestamptz`) — abreviada como `/* AUD */` para legibilidad del documento.

```sql
CREATE TABLE entidades_inmobiliarias (
  entidad_inmobiliaria_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  nombre varchar(180) NOT NULL,
  tipo_lista varchar(40) NOT NULL DEFAULT 'TIPOS_ENTIDADES_INMOBILIARIAS',

  FOREIGN KEY (tipo_lista, tipo_codigo) REFERENCES entidad,

-- Los demas datos fisicos (unidades, m2, anio, medidas...) van como ATRIBUTOS.

CREATE TABLE propiedades (
  propiedad_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  nombre varchar(180),
  tipo_lista varchar(40) NOT NULL DEFAULT 'TIPOS_PROPIEDADES',
  tipo_codigo varchar(40) NOT NULL,       -- CASA, DEPARTAMENTO, LOTE, ... (ampliable)
  entidad_inmobiliaria_id bigint REFERENCES entidades_inmobiliarias,
  empresa_id bigint REFERENCES empresas,
  ubicacion_id bigint REFERENCES ubicaciones_geograficas,
  direccion varchar(180) NOT NULL,
  precio_venta numeric(15,2) NOT NULL DEFAULT 0,      -- [Decision 6: 2 pares]
  comision_venta numeric(5,2) NOT NULL DEFAULT 0,
  precio_alquiler numeric(15,2) NOT NULL DEFAULT 0,
  comision_alquiler numeric(5,2) NOT NULL DEFAULT 0,
  estado varchar(20) NOT NULL DEFAULT 'LIBRE' CHECK (estado IN ('LIBRE','OCUPADA','VENDIDA')),
  cuenta_catastral varchar(120), numero_finca varchar(20),
  numero_lote varchar(20), numero_manzana varchar(20),   -- se mantienen: los usa la generacion de lotes
  observacion varchar(250),
  /* AUD */,
  FOREIGN KEY (tipo_lista, tipo_codigo) REFERENCES entidad
);
-- Cochera, m2, anio, dimensiones, NIS/medidores (grupo PARTE_ELECTRICA), ESSAP (grupo AGUA),
-- caracteristicas... TODOS pasan al sistema de atributos. `senado` y `resolucion` NO se migran
-- como columnas; si algun dato existe en el legado, el ETL lo vuelca como atributo.

-- ── Sistema de atributos dinámicos ──
CREATE TABLE atributos (                  -- definicion (precargada, administrable)
  atributo_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  grupo_lista varchar(40) NOT NULL DEFAULT 'GRUPOS_ATRIBUTOS',
  grupo_codigo varchar(40) NOT NULL,      -- PARTE_ELECTRICA, AGUA, ESTRUCTURA, CONFORT, ...
  nombre varchar(120) NOT NULL,           -- "Medidor trifasico", "Nro NIS", "Cantidad de pisos", "Ascensor"
  tipo_dato varchar(10) NOT NULL CHECK (tipo_dato IN ('TEXTO','NUMERO','BOOLEANO','FECHA')),
  estado varchar(10) NOT NULL DEFAULT 'ACTIVO',
  /* AUD */,
  FOREIGN KEY (grupo_lista, grupo_codigo) REFERENCES entidad,
  UNIQUE (grupo_codigo, nombre)
);

CREATE TABLE atributos_por_tipo (         -- que atributos aplican a cada tipo, y si son obligatorios
  atributo_por_tipo_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  atributo_id bigint NOT NULL REFERENCES atributos,
  tipo_lista varchar(40) NOT NULL,        -- 'TIPOS_PROPIEDADES' o 'TIPOS_ENTIDADES_INMOBILIARIAS'
  tipo_codigo varchar(40) NOT NULL,       -- EDIFICIO -> Ascensor OBLIGATORIO, Cant. pisos OBLIGATORIO
  obligatorio boolean NOT NULL DEFAULT false,
  /* AUD */,
  FOREIGN KEY (tipo_lista, tipo_codigo) REFERENCES entidad,
  UNIQUE (atributo_id, tipo_lista, tipo_codigo)
);

CREATE TABLE propiedades_atributos (      -- valores por propiedad/entidad inmobiliaria
  propiedad_atributo_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  atributo_id bigint NOT NULL REFERENCES atributos,
  propiedad_id bigint REFERENCES propiedades,
  entidad_inmobiliaria_id bigint REFERENCES entidades_inmobiliarias,
  valor text,                             -- interpretado segun atributos.tipo_dato (validado en servicio)
  /* AUD */,
  CHECK (propiedad_id IS NOT NULL OR entidad_inmobiliaria_id IS NOT NULL),
  UNIQUE (atributo_id, propiedad_id, entidad_inmobiliaria_id)
);

CREATE TABLE propietarios_entidades (     -- N:M propietarios
  propietario_entidad_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  entidad_inmobiliaria_id bigint REFERENCES entidades_inmobiliarias,
  propiedad_id bigint REFERENCES propiedades,
  propietario_id bigint NOT NULL REFERENCES socios_negocios,
  /* AUD */,
  CHECK (entidad_inmobiliaria_id IS NOT NULL OR propiedad_id IS NOT NULL)
);
```

### B.5 Dinero: documento + detalle + cobros [Decisión 8 — **alineado con la auditoría del sistema de Gestión, doc 10**]

```sql
CREATE TABLE rangos_comprobantes (        -- numeracion administrada (patron RANGO_COMPROBANTE de Gestion)

  tipo_lista varchar(40) NOT NULL DEFAULT 'TIPOS_DOCUMENTO',
  tipo_codigo varchar(40) NOT NULL,

  numero_desde bigint NOT NULL,
  numero_actual bigint NOT NULL,
  numero_hasta bigint NOT NULL,
  fecha_desde date, fecha_vencimiento date,
  timbrado varchar(25),                   -- codigo de autorizacion (SET)
  estado varchar(10) NOT NULL DEFAULT 'ACTIVO' CHECK (estado IN ('ACTIVO','INACTIVO')),
  /* AUD */,
  FOREIGN KEY (tipo_lista, tipo_codigo) REFERENCES entidad,
  UNIQUE (empresa_id, tipo_codigo, serie, numero_desde)

-- Numeracion como F_OBTENERNUMERO de Gestion: primer rango ACTIVO vigente con numero_actual < numero_hasta;
-- en PostgreSQL con SELECT ... FOR UPDATE para concurrencia (nota doc 10 §5).

CREATE TABLE documentos (                 -- comprobante universal (legal o interno, entrada o salida)
  documento_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,   -- record_id tecnico
  empresa_id bigint NOT NULL REFERENCES empresas,
  tipo_lista varchar(40) NOT NULL DEFAULT 'TIPOS_DOCUMENTO',
  tipo_codigo varchar(40) NOT NULL,       -- RECIBO, FACTURA, NOTA_INTERNA_ENTRADA/SALIDA, ...
  serie varchar(20) NOT NULL,
  numero bigint NOT NULL,
  fecha date NOT NULL,
  fecha_vencimiento date,
  socio_negocio_id bigint REFERENCES socios_negocios,
  sucursal_id bigint REFERENCES sucursales,
  moneda_id bigint NOT NULL REFERENCES monedas,
  cotizacion numeric(15,5) NOT NULL DEFAULT 1,
  total numeric(15,2) NOT NULL DEFAULT 0,
  saldo numeric(15,2) NOT NULL DEFAULT 0,             -- saldo vivo (patron Gestion)
  direccion_dinero varchar(10) NOT NULL CHECK (direccion_dinero IN ('ENTRADA','SALIDA')),
  estado varchar(20) NOT NULL DEFAULT 'PENDIENTE' CHECK (estado IN ('PENDIENTE','CANCELADO','ANULADO')),
  timbrado varchar(25),
  documento_ref_id bigint REFERENCES documentos,      -- referencia a otro comprobante (NC, etc.)
  observacion varchar(250),
  /* AUD */,
  FOREIGN KEY (tipo_lista, tipo_codigo) REFERENCES entidad,
  UNIQUE (empresa_id, tipo_codigo, serie, numero),    -- clave de negocio [Decision 8]
  CHECK (saldo >= 0 AND saldo <= total)               -- invariante de TA_DOCUMENTO
);
-- Estado derivado del saldo EN EL SERVICIO (espejo de TA_DOCUMENTO): saldo=0 -> CANCELADO; >0 -> PENDIENTE.

CREATE TABLE documentos_detalles (
  documento_detalle_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  -- record_id
  documento_id bigint NOT NULL REFERENCES documentos ON DELETE CASCADE,
  numero_item int NOT NULL,
  concepto varchar(250) NOT NULL,
  articulo bigint REFERENCES articulo,
  cantidad numeric(15,3) NOT NULL DEFAULT 1,
  precio_unitario numeric(15,2) NOT NULL DEFAULT 0,
  exentas numeric(15,2) NOT NULL DEFAULT 0,
  gravadas numeric(15,2) NOT NULL DEFAULT 0,
  porcentaje_impuesto numeric(5,2) NOT NULL DEFAULT 0,
  impuestos numeric(15,2) NOT NULL DEFAULT 0,         -- IVA incluido: g - g/(1+p/100)
  monto numeric(15,2) NOT NULL,
  saldo numeric(15,2) NOT NULL DEFAULT 0,             -- saldo por linea (para detalles_afectados, patron Gestion)
  /* AUD */,
  UNIQUE (documento_id, numero_item)
);
-- El servicio que agrega/quita detalles recalcula totales y saldo de la cabecera en la MISMA
-- transaccion (espejo de TD_DETALLE_DOCUMENTO).

CREATE TABLE planillas (                  -- caja diaria (patron PLANILLAS de Gestion) [CONFIRMADO por el usuario]
  planilla_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  empresa_id bigint NOT NULL REFERENCES empresas,
  sucursal_id bigint NOT NULL REFERENCES sucursales,
  usuario_apertura varchar(20) NOT NULL,
  fecha_apertura date NOT NULL, hora_apertura timestamptz NOT NULL,
  monto_apertura numeric(15,2) NOT NULL DEFAULT 0,
  usuario_cierre varchar(20), fecha_cierre date, hora_cierre timestamptz,
  monto_cobro numeric(15,2) NOT NULL DEFAULT 0,
  estado varchar(10) NOT NULL DEFAULT 'ABIERTA' CHECK (estado IN ('ABIERTA','CERRADA')),
  /* AUD */
);
-- Regla de Gestion: no se registran cobros sin planilla abierta (se abre automaticamente por dia/usuario/sucursal).

CREATE TABLE anulaciones (                -- log de anulaciones (patron ANULACIONES de Gestion)
  anulacion_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  empresa_id bigint NOT NULL REFERENCES empresas,
  documento_id bigint REFERENCES documentos,
  cobro_id bigint,                        -- FK se agrega tras crear cobros
  motivo_lista varchar(40) NOT NULL DEFAULT 'MOTIVOS_ANULACION', motivo_codigo varchar(40) NOT NULL,
  fecha timestamptz NOT NULL,
  /* AUD */,
  FOREIGN KEY (motivo_lista, motivo_codigo) REFERENCES entidad
);

-- ── Cobros: ESPEJO del modulo de Gestion (COBROS/COBRODETALLE/DATOS_COBROS/DETALLES_AFECTADOS) ──

CREATE TABLE cobros (                     -- espejo de COBROS de Gestion
  cobro_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  empresa_id bigint NOT NULL REFERENCES empresas,
  sucursal_id bigint NOT NULL REFERENCES sucursales,
  planilla_id bigint NOT NULL REFERENCES planillas,   -- no se cobra sin planilla abierta
  forma_pago bigint REFERENCES forma_pago,            -- FPAGO
  socio_negocio_id bigint REFERENCES socios_negocios, -- C_IDENTIDAD (quien paga)
  pagador_id bigint REFERENCES socios_negocios,       -- PAGADOR (si paga un tercero/empresa)
  cajero varchar(20),
  fecha date NOT NULL,
  hora timestamptz NOT NULL,
  monto numeric(15,2) NOT NULL DEFAULT 0,             -- acumulado por los detalles (servicio)
  moneda_id bigint NOT NULL REFERENCES monedas,
  monto_ext numeric(15,2),
  cotizacion numeric(15,5) NOT NULL DEFAULT 1,
  moneda_vuelto_id bigint REFERENCES monedas,
  vuelto numeric(15,2) NOT NULL DEFAULT 0,
  recibo_documento_id bigint REFERENCES documentos,   -- RECI generado automaticamente para creditos
  factura_documento_id bigint REFERENCES documentos,
  concepto varchar(255),
  estado varchar(10) NOT NULL DEFAULT 'ACTIVO' CHECK (estado IN ('ACTIVO','ANULADO')),  -- 'A'/'H'
  /* AUD */
);

CREATE TABLE cobros_detalles (            -- espejo de COBRODETALLE: aplicacion del cobro a un documento
  cobro_detalle_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  cobro_id bigint NOT NULL REFERENCES cobros,
  secuencia int NOT NULL,
  documento_id bigint NOT NULL REFERENCES documentos, -- el comprobante que se paga (EMPRESA_CPTE+TIPO+SERIE+NUMERO)
  monto numeric(15,2) NOT NULL,
  monto_ext numeric(15,2),
  cotizacion numeric(15,5) NOT NULL DEFAULT 1,
  estado varchar(10) NOT NULL DEFAULT 'ACTIVO' CHECK (estado IN ('ACTIVO','ANULADO')),
  /* AUD */,
  UNIQUE (cobro_id, secuencia)
);
-- Invariantes (espejo TD_COBRODETALLE / TA_COBRODETALLE, en el SERVICIO):
--   insertar detalle => documento.saldo -= monto y cobro.monto += monto;
--   anular => reverso exacto; prohibido modificar registros ANULADOS.

CREATE TABLE datos_cobros (               -- espejo de DATOS_COBROS: anexo del medio de pago
  dato_cobro_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  cobro_id bigint NOT NULL REFERENCES cobros,
  emisor_lista varchar(40) DEFAULT 'EMISORES', emisor_codigo varchar(40),        -- banco/emisor tarjeta
  procesador_lista varchar(40) DEFAULT 'PROCESADORES', procesador_codigo varchar(40),
  numero varchar(25), serie varchar(20),              -- nro de cheque/voucher
  cuenta_corriente varchar(50),
  fecha_vencimiento date,                             -- cheque diferido
  referencia varchar(50), observacion varchar(100),
  recibo varchar(20),
  cobrador_id bigint REFERENCES socios_negocios,
  fecha_deposito date, numero_deposito varchar(25),
  estado_deposito varchar(10),
  motivo_rechazo_lista varchar(40) DEFAULT 'MOTIVOS_RECHAZO', motivo_rechazo_codigo varchar(40),
  ntcr_documento_id bigint REFERENCES documentos,     -- nota de credito aplicada
  /* AUD */,
  FOREIGN KEY (emisor_lista, emisor_codigo) REFERENCES entidad,
  FOREIGN KEY (procesador_lista, procesador_codigo) REFERENCES entidad,
  FOREIGN KEY (motivo_rechazo_lista, motivo_rechazo_codigo) REFERENCES entidad
);

CREATE TABLE detalles_afectados (         -- espejo de DETALLES_AFECTADOS: a que lineas del doc se aplico el pago
  detalle_afectado_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  cobro_detalle_id bigint NOT NULL REFERENCES cobros_detalles,
  documento_detalle_id bigint NOT NULL REFERENCES documentos_detalles,
  monto numeric(15,2) NOT NULL,
  estado varchar(10) NOT NULL DEFAULT 'ACTIVO' CHECK (estado IN ('ACTIVO','ANULADO')),
  /* AUD */
);
-- Logica portada de P_DETALLES_AFECTADOS: el pago se aplica a las lineas con saldo>0 en orden (FIFO).
-- Para eso documentos_detalles lleva su propio saldo:
--   (columna agregada) documentos_detalles.saldo numeric(15,2) NOT NULL DEFAULT 0;

CREATE TABLE ingresos_egresos (           -- movimientos (comisiones, garantias, gastos)
  ingreso_egreso_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  documento_id bigint REFERENCES documentos,          -- todo movimiento con dinero real -> documento
  fecha date NOT NULL,
  tipo varchar(10) NOT NULL CHECK (tipo IN ('INGRESO','EGRESO')),
  monto numeric(15,2) NOT NULL,
  saldo numeric(15,2) NOT NULL DEFAULT 0,
  estado varchar(20) NOT NULL CHECK (estado IN ('PENDIENTE','CANCELADO','ANULADO','VENCIDO')),
  articulo bigint NOT NULL REFERENCES articulo,
  socio_negocio_id bigint REFERENCES socios_negocios,
  entidad_inmobiliaria_id bigint REFERENCES entidades_inmobiliarias,
  propiedad_id bigint REFERENCES propiedades,
  operacion_propiedad_id bigint REFERENCES operaciones_propiedades,   -- trazabilidad al contrato
  tipo_imputacion_lista varchar(40) DEFAULT 'TIPOS_IMPUTACION', tipo_imputacion_codigo varchar(40),
  forma_pago bigint REFERENCES forma_pago,
  empresa_id bigint NOT NULL REFERENCES empresas,
  observacion text,
  /* AUD */,
  FOREIGN KEY (tipo_imputacion_lista, tipo_imputacion_codigo) REFERENCES entidad
);

CREATE TABLE liquidaciones (
  liquidacion_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  operacion_propiedad_id bigint NOT NULL UNIQUE REFERENCES operaciones_propiedades,
  documento_id bigint REFERENCES documentos,          -- la liquidacion tambien emite comprobante
  fecha date NOT NULL, fecha_fiscalizacion date NOT NULL,
  entrego_llaves boolean NOT NULL DEFAULT false,
  total_garantia numeric(15,2) NOT NULL,
  total_gastos numeric(15,2) NOT NULL,
  saldo numeric(15,2) NOT NULL,
  motivo_lista varchar(40) DEFAULT 'MOTIVOS_LIQUIDACION', motivo_codigo varchar(40),
  observacion varchar(180),
  /* AUD */,
  FOREIGN KEY (motivo_lista, motivo_codigo) REFERENCES entidad
);

CREATE TABLE liquidaciones_detalles (
  liquidacion_detalle_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  liquidacion_id bigint NOT NULL REFERENCES liquidaciones ON DELETE CASCADE,
  numero_item int NOT NULL,
  articulo bigint NOT NULL REFERENCES articulo,
  monto numeric(15,2) NOT NULL,
  /* AUD */
);
```
